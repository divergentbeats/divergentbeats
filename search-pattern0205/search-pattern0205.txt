class Solution {
  public:
    vector<int> search(string &pat, string &txt) {
        int n = txt.length();
        int m = pat.length();
        vector<int> ans;
        
        for(int i=0; i<=n-m;i++)
        {
            string ch = txt.substr(i,m);
            if(ch==pat)
            ans.push_back(i);
        }
        return ans;
    }
};
